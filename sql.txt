Create table login( usernames varchar(100) not null PRIMARY KEY, passwords varchar(200) not null )
Create table CustomerType( typeid varchar(20) not null PRIMARY KEY, NameType varchar(100) not null )
Create table Customer( id varchar(20) not null PRIMARY KEY, typeid varchar(20) not null, fullname varchar(250) not null, phone varchar(20) not null, email varchar(100), CONSTRAINT fk_tid FOREIGN KEY(typeid) REFERENCES CustomerType(typeid) )
